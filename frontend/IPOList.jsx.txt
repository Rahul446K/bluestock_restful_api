import { fetchIPOs } from "../services/ipoService";

export default function IPOList() {
  const ipos = fetchIPOs();

  return (
    <div className="p-4 grid gap-4">
      {ipos.map((ipo) => (
        <div key={ipo.id} className="border p-4 rounded shadow">
          <h2 className="text-lg font-semibold">{ipo.name}</h2>
          <p>Date: {ipo.date}</p>
          <p>Price: {ipo.price}</p>
        </div>
      ))}
    </div>
  );
}
